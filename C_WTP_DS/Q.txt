void insertion_beginning()
{
	int item;
	struct node *ptr;
	ptr = (struct node *)malloc(sizeof(struct node));
	
	if(ptr==NULL)
	{
		printf("\nOverflow");
	}
	else
	{
		printf("Enter Value\n");
		scanf("%d",&item);
		
		if(head==NULL)
		{
			ptr->next = NULL;
			ptr->prev = NULL;
			ptr->data = item;
			head = ptr;
		}
		else
		{
			ptr->data = item;
			ptr->prev = NULL;
			ptr->next = head;
		    head->prev = ptr;
			head = ptr;
		}
		printf("\Node is Inserted \n");
	}
}

void insertion_last()
{
	struct node *ptr,*temp;
	int item;
	ptr = (struct node *)malloc(sizeof(struct node));
	
	if(ptr==NULL)
	{
		printf("Overflow\n");
	}
	else
	{
		printf("Enter Value\n");
		scanf("%d",&item);
		ptr->data = item;
		if(head==NULL)
		{
			ptr->next = NULL;
			ptr->prev = NULL;
			head = ptr;
		}
		else
		{
			temp = head;
			while(temp->next!=NULL)
			{
				temp= temp->next;
			}
			
			temp->next = ptr;
			ptr->prev = temp;
			ptr->next = NULL;
		}
	}
	printf("\nNode Insert at Last");
}

void insertion_specified()

					return ;

		}
		
		printf("Enter Value :");
		scanf("%d",&item);
		
		ptr->data = item;

}
